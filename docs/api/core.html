 60    def _connect(self) -> sqlite3.Connection:
 61        if self._conn is None:
 62            self._conn = sqlite3.connect(str(self.db_path))
 63            self._conn.execute("PRAGMA journal_mode=WAL")
 64            self._conn.execute("PRAGMA foreign_keys=ON")
 65            self._conn.row_factory = sqlite3.Row
 66        return self._conn

 68    def close(self) -> None:
 69        if self._conn is not None:
 70            self._conn.close()
 71            self._conn = None

 73    def _ensure_tables(self) -> None:
 74        conn = self._connect()
 75        conn.execute(_CREATE_TASK_STATE)
 76        conn.execute(_CREATE_SNAPSHOTS)
 77        # 阶段三迁移：为旧表添加 token 字段
 78        for col, default in [
 79            ("total_input_tokens", 0),
 80            ("total_output_tokens", 0),
 81            ("total_cache_hits", 0),
 82        ]:
 83            try:
 84                conn.execute(f"ALTER TABLE task_state ADD COLUMN {col} INTEGER NOT NULL DEFAULT {default}")
 85            except sqlite3.OperationalError:
 86                pass  # 列已存在
 87        conn.commit()

 91    def init_task(self, task_id: str) -> Dict[str, Any]:
 92        """初始化任务，status='input_collecting'。已存在则返回现有状态。"""
 93        conn = self._connect()
 94        existing = conn.execute(
 95            "SELECT * FROM task_state WHERE task_id = ?", (task_id,)
 96        ).fetchone()
 97        if existing is not None:
 98            return self._row_to_dict(existing)
 99
100        now = datetime.now().isoformat()
101        pipeline_json = json.dumps(_PIPELINE, ensure_ascii=False)
102        conn.execute(
103            "INSERT INTO task_state "
104            "(task_id, status, step_index, retry_count, max_retries, "
105            " user_input_json, confirmation_json, pipeline_json, created_at, updated_at) "
106            "VALUES (?, 'input_collecting', 0, 0, 3, '{}', '{}', ?, ?, ?)",
107            (task_id, pipeline_json, now, now),
108        )
109        conn.commit()
110        print(f"[StateDB] 任务初始化: {task_id}")
111        return self.get_state(task_id)

113    def get_state(self, task_id: str) -> Dict[str, Any]:
114        """返回任务完整状态 dict。"""
115        conn = self._connect()
116        row = conn.execute(
117            "SELECT * FROM task_state WHERE task_id = ?", (task_id,)
118        ).fetchone()
119        if row is None:
120            raise ValueError(f"任务不存在: {task_id}")
121        return self._row_to_dict(row)

123    def update_status(self, task_id: str, new_status: str,
124                      next_agent: Optional[str] = None) -> Dict[str, Any]:
125        """更新状态，自动计算 step_index。"""
126        conn = self._connect()
127        state = self.get_state(task_id)
128        pipeline: List[str] = json.loads(state["pipeline_json"])
129
130        step_index = 0
131        if new_status in pipeline:
132            step_index = pipeline.index(new_status)
133        elif new_status in ("completed", "cancelled"):
134            step_index = len(pipeline) - 1
135
136        now = datetime.now().isoformat()
137        conn.execute(
138            "UPDATE task_state SET status=?, step_index=?, updated_at=? "
139            "WHERE task_id=?",
140            (new_status, step_index, now, task_id),
141        )
142        conn.commit()
143        print(f"[StateDB] 状态更新: {task_id} → {new_status} (step={step_index})")
144        result = self.get_state(task_id)
145        # next_agent 是调用方传入的提示信息，不存入 DB
146        # 仅在本次返回中携带，下次 get_state() 会丢失
147        if next_agent is not None:
148            result["_next_agent_hint"] = next_agent
149        return result

151    def save_snapshot(self, task_id: str, label: str = "") -> int:

157        cursor = conn.execute(
158            "INSERT INTO snapshots (task_id, state_json, label, created_at) "

162        conn.commit()

167    def rollback(self, task_id: str, steps: int = 1) -> bool:
168        """回滚到倒数第 N 个快照。"""
169        conn = self._connect()
170        rows = conn.execute(
171            "SELECT id, state_json FROM snapshots WHERE task_id = ? "
172            "ORDER BY id DESC LIMIT ?",
173            (task_id, steps),
174        ).fetchall()
175
176        if not rows:
177            print(f"[StateDB] 回滚失败: 无快照 ({task_id})")
178            return False
179
180        # 取倒数第 N 个（即列表最后一个元素）
181        target = rows[-1]
182        restored = json.loads(target["state_json"])
183
184        now = datetime.now().isoformat()
185        conn.execute(
186            "UPDATE task_state SET status=?, step_index=?, retry_count=?, "
187            "user_input_json=?, confirmation_json=?, updated_at=? "
188            "WHERE task_id=?",
189            (
190                restored["status"],
191                restored["step_index"],
192                restored["retry_count"],
193                restored.get("user_input_json", "{}"),
194                restored.get("confirmation_json", "{}"),
195                now,
196                task_id,
197            ),
198        )
199        conn.commit()
200        print(f"[StateDB] 回滚成功: {task_id} → {restored['status']} (snapshot id={target['id']})")
201        return True

 25class Orchestrator:
 26    """Python 编排器，外层驱动管线流转。
 27
 28    阶段二新增：验证、重试、质量门。
 29    """
 30
 31    def __init__(self, task_dir: str, task_id: str,
 32                 project_root: str = ".",
 33                 max_retries: int = 3) -> None:
 34        self.task_id = task_id
 35        self.task_dir = task_dir
 36        self.db = StateDB(task_dir)
 37        self.caller = AgentCaller.create(project_root)
 38        self.gate_runner = QualityGateRunner()
 39        self.max_retries = max_retries
 40
 41        # 阶段二状态
 42        self._last_validation: Tuple[bool, str] = (True, "")
 43        self._last_gate_result: Dict[str, Any] = {"approved": True, "results": []}
 44        self._force_fail_step: Optional[str] = None
 45
 46        # 阶段三：token 监控
 47        self._total_input_tokens: int = 0
 48        self._total_output_tokens: int = 0
 49
 50    # ---- 阶段二公开查询接口 ----
 51
 52    def get_validation_status(self) -> Tuple[bool, str]:
 53        """返回最近一次验证结果 (passed, message)。"""
 54        return self._last_validation
 55
 56    def get_retry_count(self) -> int:
 57        """从 DB 返回当前重试次数。"""
 58        return self.db.get_state(self.task_id).get("retry_count", 0)
 59
 60    def force_fail_step(self, step_name: str) -> None:
 61        """用于测试：强制指定步骤验证失败。"""
 62        self._force_fail_step = step_name
 63
 64    # ---- 主循环 ----
 65
 66    def run(self) -> Dict[str, Any]:
 67        """遍历 PIPELINE，自动执行 needs_user=False 的阶段。
 68
 69        阶段三升级：
 70        - agent 返回 tool_calls 时走 Tool Use 审批流程
 71        - 每次 agent 调用后累加 token 用量到 db
 72        """
 73        state = self.db.get_state(self.task_id)
 74        current_status = state["status"]
 75
 76        idx = 0
 77        while idx < len(PIPELINE):
 78            status, agent_id, needs_user = PIPELINE[idx]
 79
 80            # 跳过已完成的阶段
 81            step_index = state["step_index"]
 82            stage_index = PIPELINE_STEPS.index(status)
 83            if stage_index < step_index:
 84                idx += 1
 85                continue
 86
 87            # 需要用户输入的阶段 → 暂停等待
 88            if needs_user:
 89                print(f"[Orchestrator] 暂停于 {status}，等待用户输入")
 90                return {
 91                    "status": status,
 92                    "waiting_for": "user_input" if agent_id else "confirmation",
 93                    "step_index": stage_index,
 94                }
 95
 96            # 自动阶段 → 调用 agent
 97            if agent_id is not None:
 98                self.db.save_snapshot(self.task_id, label=f"before_{status}")

346    def _handle_tool_call(self, current_step: str,
347                          tool_call: Dict[str, Any]) -> bool:
348        """处理单个 transition_state tool call。
349
350        合法转换 → 验证输出 → 批准
351        非法转换（跳步） → 拒绝
352        """
353        inp = tool_call.get("input", {})
354        next_step = inp.get("next_step", "")
355        summary = inp.get("output_summary", "")
356        errors = inp.get("errors", [])
357
358        # 检查是否合法转换（不跳步）
359        if next_step not in PIPELINE_STEPS:
360            print(f"[Orchestrator] Tool Use 拒绝: 未知步骤 '{next_step}'")
361            self._feedback_to_agent(current_step,
362                                    f"非法步骤 '{next_step}'，可选: {PIPELINE_STEPS}")
363            return False
364
365        current_idx = PIPELINE_STEPS.index(current_step)
366        target_idx = PIPELINE_STEPS.index(next_step)
367
368        if target_idx > current_idx + 1:
369            print(f"[Orchestrator] Tool Use 拒绝: 跳步 "
370                  f"{current_step}({current_idx}) → {next_step}({target_idx})")
371            self._feedback_to_agent(current_step,
372                                    f"不允许从 {current_step} 跳到 {next_step}，"
373                                    f"只能推进到下一步")
374            return False
375
376        # 合法转换 → 验证当前步骤输出
377        passed, msg = self._validate_step(current_step)
378        self._last_validation = (passed, msg)
379
380        if not passed:
381            print(f"[Orchestrator] Tool Use 验证失败: {msg}")
382            self._feedback_to_agent(current_step, f"验证失败: {msg}")
383            self._handle_validation_failure(current_step, msg)
384            return False
385
386        # 验证通过 → 批准转换
387        target_agent = dict((s, a) for s, a, _ in PIPELINE).get(next_step)
388        self.db.update_status(self.task_id, next_step, target_agent)
389        print(f"[Orchestrator] Tool Use 批准: {current_step} → {next_step} "
390              f"| summary: {summary[:80]}")
391        return True

155                if retry_count >= self.max_retries:
156                    self.db.update_status(self.task_id, "failed")
157                    print(f"[Orchestrator] 验证重试耗尽，任务失败: {self.task_id}")
158                    return {"status": "failed", "step": status, "error": msg}
159                # 重新获取 state（回退后状态已变）
160                state = self.db.get_state(self.task_id)
161                current_status = state["status"]
162                # 如果回退了阶段，idx 也要回退到对应位置
163                if current_status in PIPELINE_STEPS:
164                    new_idx = PIPELINE_STEPS.index(current_status)
165                    if new_idx < stage_index:
166                        idx = new_idx
167                        continue
168                continue
169
170            # 验证通过，重置重试计数
171            self.db.reset_retry_count(self.task_id)
172
173            # --- 阶段二：质量门检查 ---
174            gate_result = self.gate_runner.run_gates(
175                Path(self.task_dir), status
176            )
177            self._last_gate_result = gate_result
178
179            if not gate_result["approved"]:
180                self._handle_gate_failure(status, gate_result)
181                retry_count = self.db.get_state(self.task_id).get("retry_count", 0)
182                if retry_count >= self.max_retries:
183                    self.db.update_status(self.task_id, "failed")
184                    print(f"[Orchestrator] 质量门重试耗尽，任务失败: {self.task_id}")
185                    return {"status": "failed", "step": status, "gate": gate_result}
186                state = self.db.get_state(self.task_id)
187                current_status = state["status"]
188                # 如果回退了阶段，idx 也要回退
189                if current_status in PIPELINE_STEPS:
190                    new_idx = PIPELINE_STEPS.index(current_status)
191                    if new_idx < stage_index:
192                        idx = new_idx
193                        continue
194                continue
195
196            # 推进到下一阶段
197            next_idx = stage_index + 1
198            if next_idx < len(PIPELINE):
199                next_status, next_agent, _ = PIPELINE[next_idx]
200                self.db.update_status(self.task_id, next_status, next_agent)
201                print(f"[Orchestrator] {status} → {next_status} | Agent: {next_agent}")
202            else:
203                self.db.update_status(self.task_id, "archiving")
204                print(f"[Orchestrator] {status} → archiving (最终阶段)")
205
206            state = self.db.get_state(self.task_id)
207            idx += 1
208
209        return {"status": "archiving", "step_index": len(PIPELINE) - 1}
